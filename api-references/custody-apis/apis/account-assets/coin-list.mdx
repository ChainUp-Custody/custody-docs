---
title: 获取币种列表
api: GET /api/v2/user/getCoinList
description: "获取商户的币种列表"
---

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>



#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="symbol" type="string">币种（调用提币的接口，及任何查询接口时务必使用此字段返回的值）</ResponseField>
        <ResponseField name="icon" type="string">币种icon</ResponseField>
        <ResponseField name="real_symbol" type="string">币种链上名称</ResponseField>
        <ResponseField name="decimals" type="string">精度</ResponseField>
        <ResponseField name="name" type="string">币种全称</ResponseField>
        <ResponseField name="base_symbol" type="string">主链币币名</ResponseField>
        <ResponseField name="contract_address" type="string">合约地址</ResponseField>
        <ResponseField name="deposit_confirmation" type="string">币种充值确认数</ResponseField>
        <ResponseField name="explorer" type="string">区块浏览器</ResponseField>
        <ResponseField name="address_regex" type="string">地址正则</ResponseField>
        <ResponseField name="address_tag_regex" type="string">tag正则</ResponseField>
        <ResponseField name="support_memo" type="string">是否支持memo，0不支持1支持</ResponseField>
        <ResponseField name="support_token" type="string">是否支持token币，0不支持1支持,主链币才有值,代币为空</ResponseField>
        <ResponseField name="address_link" type="string">区块浏览器地址查询链接前缀</ResponseField>
        <ResponseField name="tx_link" type="string">区块浏览器交易查询链接前缀</ResponseField>
        <ResponseField name="merge_address_symbol" type="string">合并地址主链币，币种唯一标识</ResponseField>
    </Expandable>
</ResponseField>
