---
title: 批量查询商户转账记录
description: "获取统一托管钱包商户特定ID的转账记录"
---

`GET /api/v2/account/transferList`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>

<ParamField query="ids" type="string" required>
    请求唯一标识,多个之间用英文逗号分割，最多100个
</ParamField>
<ParamField query="ids_type" type="string" required>
    request_id：请求ID（默认）；receipt：转账凭证
</ParamField>


#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="time" type="long">当前时间戳</ResponseField>
        <ResponseField name="charset" type="string">编码格式，无特殊情况，传参数`utf-8`</ResponseField>
        <ResponseField name="version" type="string">接口版本号，无特殊情况，传参数`v2`</ResponseField>
        <ResponseField name="id" type="string">请求唯一标识，最多支持64位</ResponseField>
        <ResponseField name="symbol" type="string">币种</ResponseField>
        <ResponseField name="amount" type="string">转账数量，包含转账手续费</ResponseField>
        <ResponseField name="from" type="string">转出商户`APPID`</ResponseField>
        <ResponseField name="to" type="string">转入商户`APPID`</ResponseField>
        <ResponseField name="created_at" type="long">创建时间</ResponseField>
        <ResponseField name="request_id" type="string">三方ID</ResponseField>
        <ResponseField name="receipt" type="string">转账凭证</ResponseField>
        <ResponseField name="remark" type="string">最大支持32字符</ResponseField>
    </Expandable>
</ResponseField>
