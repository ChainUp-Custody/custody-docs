title: Custody商户转账
description: "Keysecure内部商户互相转账"
---

`POST /api/v2/account/transfer`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>


<ParamField query="request_id" type="string" required>
    请求唯一标识，最多支持64位
</ParamField>
<ParamField query="symbol" type="string" required>
    转账币种，Keysecure获取币种名称
</ParamField>
<ParamField query="amount" type="string" required>
    转账数量，包含转账手续费
</ParamField>
<ParamField query="to" type="string" required>
    转入商户，目前填写商户`APPID`
</ParamField>
<ParamField query="remark" type="string">
    备注字段
</ParamField>


#### 响应参数数据结构
<Expandable title="object">
    <ResponseField name="receipt" type="string">转账唯一凭证</ResponseField>
</Expandable>
