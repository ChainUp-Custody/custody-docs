---
title: 同步充值记录
description: "获取统一托管钱包的特定ID后的收款记录"
---

`GET /api/v2/billing/syncDepositList`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>

<ParamField query="max_id" type="string" required>
    返回大于id的100条充值记录数据
</ParamField>


#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">充值唯一id</ResponseField>
        <ResponseField name="uid" type="integer">充值用户id</ResponseField>
        <ResponseField name="symbol" type="string">币种</ResponseField>
        <ResponseField name="amount" type="string">充值金额</ResponseField>
        <ResponseField name="created_at" type="long">创建时间</ResponseField>
        <ResponseField name="updated_at" type="long">修改时间</ResponseField>
        <ResponseField name="txid" type="string">区块链交易ID</ResponseField>
        <ResponseField name="confirmations" type="integer">区块链确认数</ResponseField>
        <ResponseField name="address_from" type="string">来源地址</ResponseField>
        <ResponseField name="address_to" type="string">充值到帐地址</ResponseField>
        <ResponseField name="status" type="integer">0待确认，1 成功，2 失败，4 待KYT验证，5 待人工审核(KYT风险等级过高)，6 待人工审核(KYT充值熔断)，7 已转入</ResponseField>
    </Expandable>
</ResponseField>
