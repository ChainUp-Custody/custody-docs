---
title: 获取归集矿工费
description: "账户类型的币种充值后需要进行归集，UTXO无归集矿工费费用"
---

`GET /api/v2/billing/syncMinerFeeList`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>

<ParamField query="max_id" type="integer">
    返回大于 id 的 100 条归集矿 工费记录数据
</ParamField>


#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">归集唯一ID</ResponseField>
        <ResponseField name="symbol" type="string">币种</ResponseField>
        <ResponseField name="amount" type="string">归集金额</ResponseField>
        <ResponseField name="fee" type="string">归集手续费</ResponseField>
        <ResponseField name="created_at" type="long">创建时间</ResponseField>
        <ResponseField name="updated_at" type="long">修改时间</ResponseField>
        <ResponseField name="txid" type="string">区块链交易ID</ResponseField>
        <ResponseField name="confirmations" type="integer">区块链确认数</ResponseField>
        <ResponseField name="status" type="integer">0 待确认，1 成功，2 失败，4 待KYT验证，5 待人工审核(KYT风险等级过高)，6 待人工审核(KYT充值熔断)，7 已转入。[详细请参考](/help-center/manage-joint/account-transfer/transfer#ii)</ResponseField>
        <ResponseField name="address_from" type="string">归集发送地址</ResponseField>
        <ResponseField name="address_to" type="string">归集到帐地址</ResponseField>
        <ResponseField name="txid_type" type="string">0 链上交易，1 联盟转账交易</ResponseField>
        <ResponseField name="base_symbol" type="string">原生代币名称</ResponseField>
        <ResponseField name="contract_address" type="string">币种合约地址</ResponseField>
        <ResponseField name="email" type="string">邮箱</ResponseField>
    </Expandable>
</ResponseField>
