---
title: 批量获取提现记录
description: "获取统一托管钱包特定ID的提现记录"
---

`GET /api/v2/billing/withdrawList`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>

<ParamField query="ids" type="string" required>
    多个充值request_id使用逗号隔开，最多100个request_id
</ParamField>


#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="request_id" type="string">请求id</ResponseField>
        <ResponseField name="id" type="integer">提现id</ResponseField>
        <ResponseField name="uid" type="integer">提现用户id</ResponseField>
        <ResponseField name="symbol" type="string">币种</ResponseField>
        <ResponseField name="amount" type="string">提现金额</ResponseField>
        <ResponseField name="withdraw_fee_symbol" type="string">提现手续费币种</ResponseField>
        <ResponseField name="withdraw_fee" type="string">提现手续费</ResponseField>
        <ResponseField name="fee_symbol" type="string">挖矿手续费币种</ResponseField>
        <ResponseField name="real_fee" type="string">矿工费</ResponseField>
        <ResponseField name="created_at" type="long">创建时间</ResponseField>
        <ResponseField name="updated_at" type="long">修改时间</ResponseField>
        <ResponseField name="address_from" type="string">来源地址</ResponseField>
        <ResponseField name="address_to" type="string">到账地址</ResponseField>
        <ResponseField name="txid" type="string">区块链交易ID</ResponseField>
        <ResponseField name="confirmations" type="integer">区块链确认数</ResponseField>
        <ResponseField name="saas_status" type="integer">平台审核状态</ResponseField>
        <ResponseField name="company_status" type="integer">商户审核状态</ResponseField>
        <ResponseField name="status" type="integer">提现状态: 0 风控审核中，1 支付中，2 审核拒绝，4 失败，5 成功， 6 已撤销，7 待KYT验证， 8 待人工审核(KYT风险等级过高)。[详细请参考](/help-center/manage-joint/account-transfer/transfer#ii)</ResponseField>
    </Expandable>
</ResponseField>
