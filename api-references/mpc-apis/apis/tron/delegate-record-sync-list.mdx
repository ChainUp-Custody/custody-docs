---
title: 同步tron购买资源交易记录
api: GET /api/mpc/tron/delegate/sync_trans_list
description: "获取工作站下所有钱包特定交易ID之后的TRON交易资源购买记录"
---

#### 请求参数数据结构
<ParamField query="max_id" type="string" required>
    TRON购买资源交易记录起始id，例100，系统将返回101后的所有交易信息
</ParamField>



#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">购买资源交易ID</ResponseField>
        <ResponseField name="request_id" type="string">交易唯一标识</ResponseField>
        <ResponseField name="address_form" type="integer">资源接受地址</ResponseField>
        <ResponseField name="service_charge_rate" type="string">服务费费率</ResponseField>
        <ResponseField name="service_charge" type="string">服务费 单位TRX</ResponseField>
        <ResponseField name="contract_address" type="string">合约地址 用于预估矿工费</ResponseField>
        <ResponseField name="address_to" type="string">转账到账地址 用于预估矿工费</ResponseField>
        <ResponseField name="resource_type" type="string">购买资源类型 0:能量与带宽 1:能量</ResponseField>
        <ResponseField name="buy_type" type="string">购买资源方式 0:系统预估 1:指定数量</ResponseField>
        <ResponseField name="net_num" type="string">代理带宽数量</ResponseField>
        <ResponseField name="energy_num" type="long">代理能量数量</ResponseField>
        <ResponseField name="net_txid" type="long">带宽代理交易txid</ResponseField>
        <ResponseField name="energy_txid" type="string">能量代理交易txid</ResponseField>
        <ResponseField name="net_price" type="string">带宽价格</ResponseField>
        <ResponseField name="energy_price" type="string">能量价格</ResponseField>
        <ResponseField name="status" type="integer">状态 0创建订单成功1代理资源成功2代理资源失败</ResponseField>
    </Expandable>
</ResponseField>
