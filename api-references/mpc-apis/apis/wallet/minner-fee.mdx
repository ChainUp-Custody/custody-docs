---
title: "预估转账消耗的矿工费"
api: "GET /v2/chain/minerfee"
description: "预估当前币种转账消耗的矿工费"
---

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳，例: `1600128590`
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数`utf-8`
</ParamField>
<ParamField query="version" type="string" required>
    版本，无特殊情况，传参数`v2`
</ParamField>
<ParamField query="symbol" type="string" required>
    币种唯一标识，转账时使用，例：`USDTERC20`
</ParamField>

#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="feeSymbol" type="string">手续费币种，例：`BSC`</ResponseField>
        <ResponseField name="symbol" type="string">币种，唯一标识，例：`USDTERC20`</ResponseField>
        <ResponseField name="fee" type="string">手续费，例：`0.00001`</ResponseField>
        <ResponseField name="type" type="integer">类型，`0`主链币转账，`1`代币转账</ResponseField>
        <ResponseField name="ifCollection" type="bool">币种是否支持归集，`true`支持，`false`不支持</ResponseField>
    </Expandable>
</ResponseField>
