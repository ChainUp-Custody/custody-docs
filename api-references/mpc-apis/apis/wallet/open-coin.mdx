#### 响应参数数据结构
<ResponseField name="open_main_chain" type="object[ ]">
    工作站开通的MPC网络信息
    <Expandable title="object">
        <ResponseField name="coin_net" type="string">币种网络，例：`MATIC`</ResponseField>
        <ResponseField name="symbol" type="string">币种信息唯一标识，例：`MATIC1`</ResponseField>
        <ResponseField name="symbol_alias" type="string">币种别名，例：`MATIC`</ResponseField>
        <ResponseField name="support_acceleration" type="bool">币种是否支持加速，`ture`支持，`false`不支持</ResponseField>
    </Expandable>
</ResponseField>
<ResponseField name="support_main_chain" type="object[ ]">
    CryptoSafe支持的MPC网络信息
    <Expandable title="object">
        <ResponseField name="coin_net" type="string">币种网络，例：`ETH`</ResponseField>
        <ResponseField name="symbol" type="string">币种信息唯一标识，例：`ETH`</ResponseField>
        <ResponseField name="symbol_alias" type="string">币种别名，例：`ETH`</ResponseField>
        <ResponseField name="support_acceleration" type="bool">币种是否支持加速，`ture`支持，`false`不支持</ResponseField>
        <ResponseField name="if_open_chain" type="bool">钱包是否开通网络，`ture`支持，`false`不支持</ResponseField>
    </Expandable>
</ResponseField>
