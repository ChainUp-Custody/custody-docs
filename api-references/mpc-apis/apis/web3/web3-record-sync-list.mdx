---
title: 同步web3交易记录
api: GET /api/mpc/web3/sync_trans_list
description: "获取工作站下所有钱包web3交易记录，最多返回100条"
---

#### 请求参数数据结构
<ParamField query="max_id" type="string" required>
    Web3交易记录起始id
</ParamField>



#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Web3交易ID</ResponseField>
        <ResponseField name="request_id" type="string">交易唯一标识</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">钱包id</ResponseField>
        <ResponseField name="txid" type="string">交易hash</ResponseField>
        <ResponseField name="symbol" type="string">交易币种</ResponseField>
        <ResponseField name="main_chain_symbol" type="string">主链币币名（唯一标识），例：`ETH`</ResponseField>
        <ResponseField name="amount" type="string">交易金额。-1表示无穷大</ResponseField>
        <ResponseField name="fee_symbol" type="string">矿工费币种，例：ETH</ResponseField>
        <ResponseField name="fee" type="string">矿工费，例：0.00123</ResponseField>
        <ResponseField name="real_fee" type="string">真实消耗的矿工费，例：0.00111</ResponseField>
        <ResponseField name="created_at" type="long">创建时间时间戳</ResponseField>
        <ResponseField name="updated_at" type="long">修改时间时间戳</ResponseField>
        <ResponseField name="address_from" type="string">交易from地址</ResponseField>
        <ResponseField name="address_to" type="string">交易to地址</ResponseField>
        <ResponseField name="interactive_contract" type="string">交互合约</ResponseField>
        <ResponseField name="confirmations" type="integer">区块确认数，例：10</ResponseField>
        <ResponseField name="input_data" type="string">合约交易的方法参数组成的16进制数据</ResponseField>
        <ResponseField name="status" type="integer">交易状态: 1100 待签名，1200 支付中，2000 支付完成，2100 审批拒绝，2200 拒绝，2300 交易丢弃，2400 支付失败</ResponseField>
        <ResponseField name="trans_source" type="integer">交易类型： `1`app, `2`openapi</ResponseField>
        <ResponseField name="tx_height" type="long">交易完成的区块高度</ResponseField>
    </Expandable>
</ResponseField>
