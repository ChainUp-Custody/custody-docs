---
title: 获取转账记录
api: GET /api/mpc/billing/withdraw_list
description: "获取工作站下所有钱包特定ID的转账记录"
---

#### 请求参数数据结构
<ParamField query="ids" type="string" required>
    多`request_id`的字符串，英文逗号分割
</ParamField>



#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">转账id</ResponseField>
        <ResponseField name="request_id" type="string">转账唯一标识</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">钱包id</ResponseField>
        <ResponseField name="symbol" type="string">币种唯一标识，转账时使用，例：`USDTERC20`</ResponseField>
        <ResponseField name="amount" type="string">转账金额</ResponseField>
        <ResponseField name="fee_symbol" type="string">矿工费币种，例：ETH</ResponseField>
        <ResponseField name="fee" type="string">矿工费，例：0.00123</ResponseField>
        <ResponseField name="real_fee" type="string">真实消耗的矿工费，例：0.00111</ResponseField>
        <ResponseField name="created_at" type="long">创建时间时间戳</ResponseField>
        <ResponseField name="updated_at" type="long">修改时间时间戳</ResponseField>
        <ResponseField name="address_from" type="string">出币地址</ResponseField>
        <ResponseField name="address_to" type="string">到账地址</ResponseField>
        <ResponseField name="memo" type="string">转账到账地址memo</ResponseField>
        <ResponseField name="txid" type="string">交易hash</ResponseField>
        <ResponseField name="confirmations" type="integer">区块确认数，例：10</ResponseField>
        <ResponseField name="status" type="integer">转账状态: 1000 未审批，1010 KYT验证中状态，1020 待KYT人工处理，1100 审批通过\待签名, 1200 支付中，2000 支付完成，2100 审批拒绝，2101 KYT验证拒绝，2200 拒绝, 2201 系统拒绝，2300 交易丢弃，2400 支付失败。[详细请参考](/help-center/manage-mpc/transaction/transfer#ii)</ResponseField>
        <ResponseField name="withdraw_source" type="integer">转账类型： `1`app, `2`openapi, `3`web</ResponseField>
        <ResponseField name="base_symbol" type="string">转账币种网络唯一标识，例：ETH</ResponseField>
        <ResponseField name="contract_address" type="string">转账币种合约地址</ResponseField>
        <ResponseField name="tx_height" type="long">交易完成的区块高度</ResponseField>
    </Expandable>
</ResponseField>
