`GET /api/v2/account/getDepositAddressInfo`

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    当前时间戳
</ParamField>
<ParamField query="charset" type="string" required>
    编码格式，无特殊情况，传参数utf-8
</ParamField>
<ParamField query="version" type="string" required>
    接口版本号，无特殊情况，传参数v2
</ParamField>


<ParamField query="address" type="string" required>
    任意地址
</ParamField>


#### 响应参数数据结构
<Expandable title="object">

    <ResponseField name="symbol" type="string">币种唯一标识</ResponseField>
