---
title: Get stake record list
api: GET /api/mpc/babylon/stake_list
description: "Get stake records greater than a specific stake_id, returning 20 pieces of data each time"
---

#### Request Parameters
<ParamField query="max_id" type="string" required>
    The starting stake_id of the stake record, for example 100, the system will return the 20 transaction information after 100
</ParamField>


#### Response Data Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Platform stake_id</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, e.g.: `BTC`</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">Wallet ID</ResponseField>
        <ResponseField name="updated_at" type="long">Modified timestamp</ResponseField>
        <ResponseField name="request_id" type="string">The unique identifier for stake</ResponseField>
        <ResponseField name="fp_key" type="string">Finality Provider</ResponseField>
        <ResponseField name="status" type="integer">Stake status: 0 Staking, 1000 Stake tx confirmed but confirmation < stake lock number, 2000 Stake successful, 2100 Funds locked but stake failed, 3000 Unbonding, 3100 Unbond tx confirmed but confirmation < unbond lock number, 4000 Stake withdrawable, 4100 Unbond Withdrawable, 4200 Unstaking, 4300 Unbond unstaking, 5000 Stake done, 5100 Stake cancel</ResponseField>
        <ResponseField name="stake" type="object">
            Stake Info
            <Expandable title="object">
                <ResponseField name="address_from" type="string">Address from</ResponseField>
                <ResponseField name="address_to" type="string">Address to</ResponseField>
                <ResponseField name="fee" type="string">Mining fee, e.g.ï¼š0.00123</ResponseField>
                <ResponseField name="amount" type="string">Amount</ResponseField>
                <ResponseField name="created_at" type="long">Creation time timestamp</ResponseField>
                <ResponseField name="txid" type="string">Transaction hash</ResponseField>
                <ResponseField name="tx_height" type="long">Block height at which the transaction is completed</ResponseField>
                <ResponseField name="tx_type" type="integer">1 Stake 2 Unbond 3 Unstake 4 Unbound unstake </ResponseField>
                <ResponseField name="lock_number" type="integer">Staking period (block height)</ResponseField>
                <ResponseField name="remark" type="string">Remark</ResponseField>
           </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>
