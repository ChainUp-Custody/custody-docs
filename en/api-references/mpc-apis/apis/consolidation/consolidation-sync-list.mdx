---
title: Sync Consolidation Records
api: GET /api/mpc/billing/sync_auto_collect_list
description: "Retrieve up to 100 consolidation records for all sub-wallets under a wallet."
---

#### Request Parameters
<ParamField query="max_id" type="string" required>
    Starting ID for consolidation records
</ParamField>



#### Response Data Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Consolidation ID</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">Sub wallet id</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, used during consolidation, e.g.：`USDTERC20`</ResponseField>
        <ResponseField name="amount" type="string">Consolidation amount</ResponseField>
        <ResponseField name="fee_symbol" type="string">Fee currency, e.g.：ETH</ResponseField>
        <ResponseField name="fee" type="string">Fee amount, e.g.：0.00123</ResponseField>
        <ResponseField name="real_fee" type="string">Actual consumed fee, e.g.：0.00111</ResponseField>
        <ResponseField name="created_at" type="long">Creation timestamp</ResponseField>
        <ResponseField name="updated_at" type="long">Modified timestamp</ResponseField>
        <ResponseField name="address_from" type="string">Sender's address</ResponseField>
        <ResponseField name="address_to" type="string">Consolidation address</ResponseField>
        <ResponseField name="txid" type="string">Transaction hash</ResponseField>
        <ResponseField name="confirmations" type="integer">Number of block confirmations, e.g.：10</ResponseField>
        <ResponseField name="status" type="integer">Consolidation status: 1000 Unapproved, 1100 Approved, Pending Signature, 1200 In Progress, 2000 Completed, 2100 Rejected, 2200 Rejected, 2201 System Rejected, 2202 Auto Cancelled, 2300 Transaction Discarded, 2400 Payment Failed</ResponseField>
        <ResponseField name="trans_type" type="integer">Consolidation type: 10. Consolidation Transaction, 11. Consolidation Gas Transaction</ResponseField>
        <ResponseField name="base_symbol" type="string">Base currency unique identifier on the main chain, e.g.：ETH</ResponseField>
        <ResponseField name="contract_address" type="string">Contract address for the consolidation currency</ResponseField>
    </Expandable>
</ResponseField>
