---
title: Get Receiving records
api: GET /api/mpc/billing/deposit_list
description: "Get all wallet receiving records under the workspace"
---

#### Request Parameters
<ParamField query="ids" type="string" required>
    A string of multiple ids, separated by commas, e.g.：`123,345`
</ParamField>



#### Response Data Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Receiving id</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">Wallet id</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, used for transfers, e.g.：`USDTERC20`</ResponseField>
        <ResponseField name="created_at" type="long">Creation timestamp</ResponseField>
        <ResponseField name="updated_at" type="long">Modified timestamp</ResponseField>
        <ResponseField name="amount" type="string">receiving amount</ResponseField>
        <ResponseField name="address_from" type="string">transfer from address</ResponseField>
        <ResponseField name="address_to" type="string">receiving address</ResponseField>
        <ResponseField name="memo" type="string">receiving address memo</ResponseField>
        <ResponseField name="txid" type="string">Transaction hash</ResponseField>
        <ResponseField name="confirmations" type="integer">Number of block confirmations, e.g.：10</ResponseField>
        <ResponseField name="status" type="integer">Receiving status: `1000` Unconfirmed, `1010` Pending KYT Verification, `1020` Pending Manual Review, `1030` Deposit KYT Circuit Breaker (Pending Manual Review), `1800` Transaction Retrace, `1900` Withdrawal Unlock, `2000` Completed (Credited Completed), `3000` Abnormal, `4000` Returned. [Details refer to](/help-center/manage-mpc/transaction/transfer#iii)</ResponseField>
        <ResponseField name="deposit_type" type="integer">Deposit Type, 1: Regular coin deposit, 2: NFT deposit</ResponseField>
        <ResponseField name="token_id" type="string">TokenId for NFT deposit, this value is not empty when: `deposit_type=2`</ResponseField>
        <ResponseField name="base_symbol" type="string">The unique identifier of the network of the coin to be received, e.g.：ETH</ResponseField>
        <ResponseField name="contract_address" type="string">received coin's contract address</ResponseField>
        <ResponseField name="tx_height" type="long">Block height at which the transaction is completed</ResponseField>
        <ResponseField name="refund_amount" type="string">Returned amount</ResponseField>
    </Expandable>
</ResponseField>
