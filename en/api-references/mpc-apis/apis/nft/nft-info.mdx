---
title: Get NFT Information
api: POST /api/nft/info
description: "Retrieve NFT data such as images and names."
---

#### Request Parameters
<ParamField query="base_symbol" type="long" required>
    Native Token of the Networks, Unique identifier for the coin, e.g.：`ETH`
</ParamField>
<ParamField query="contract_address" type="string" required>
    NFT contract address
</ParamField>
<ParamField query="token_id" type="string">
    NFT token ID
</ParamField>



#### Response Data Parameters
<ResponseField name="base_symbol" type="string">Native Token of the Networks, Unique identifier for the coin, e.g.：`ETH`</ResponseField>
<ResponseField name="contract_address" type="string">NFT contract address</ResponseField>
<ResponseField name="token_id" type="string">NFT token ID</ResponseField>
<ResponseField name="nft_name" type="string">NFT name, when absent, use "# + tokenId", e.g.: #1000001</ResponseField>
<ResponseField name="image" type="string">URL or base64 of NFT image or video</ResponseField>
<ResponseField name="image_type" type="string">NFT image or video type, e.g.：".png", ".mp4"</ResponseField>
<ResponseField name="is_video" type="bool">Indicates if it's a video. `true` if it's a video, `false` otherwise</ResponseField>
