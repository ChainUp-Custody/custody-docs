---
title: Purchase Tron Resource Trading
api: POST /api/mpc/tron/delegate
description: "Purchase Tron resources through API"
---

#### Request Parameters
<ParamField query="request_id" type="string" required>
    The unique identifier for transferring coins
</ParamField>
<ParamField query="resource_type" type="integer" >
    Purchase resource type 0: Energy and bandwidth 1: Energy not specified defaults to 0
</ParamField>
<ParamField query="buy_type" type="integer">
    Method of purchasing resources 0: System estimation 1: Specify quantity not specified, default is 0
</ParamField>
<ParamField query="net_num" type="integer" >
    Number of bandwidths
</ParamField>
<ParamField query="energy_num" type="integer" >
    Quantity of energy
</ParamField>
<ParamField query="address_from" type="string" required>
    Address for receiving resources
</ParamField>
<ParamField query="address_to" type="string">
    Transfer to account address for estimating miner fees
</ParamField>
<ParamField query="contract_address" type="string">
    The contract address is used to estimate miner fees
</ParamField>



#### Response Data Parameters
<Expandable title="object">
    <ResponseField name="trans_id" type="integer">Purchase Resource Transaction ID</ResponseField>
</Expandable>
