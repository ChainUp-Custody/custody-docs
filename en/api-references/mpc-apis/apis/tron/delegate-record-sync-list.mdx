---
title: Synchronize Tron purchasing resource transaction records
api: GET /api/mpc/tron/delegate/sync_trans_list
description: "Obtain the record of TRON transaction resource purchases following a specific transaction ID for all wallets under the workstation."
---

#### Request Parameters
<ParamField query="max_id" type="string" required>
    The starting ID for TRON resource purchase transaction records, for example, 100, where the system will return all transaction information following ID 101.
</ParamField>



#### Response Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Resource purchase transaction ID</ResponseField>
        <ResponseField name="request_id" type="string">Unique transaction identifier</ResponseField>
        <ResponseField name="receive_address" type="integer">Resource receiving address</ResponseField>
        <ResponseField name="service_charge_rate" type="string">Service fee rate</ResponseField>
        <ResponseField name="service_charge" type="string">Service fee in TRX</ResponseField>
        <ResponseField name="contract_address" type="string">Contract address for estimating miner fees</ResponseField>
        <ResponseField name="address_to" type="string">Transfer destination address for estimating miner fees</ResponseField>
        <ResponseField name="resource_type" type="string">Resource purchase type: 0 for Energy and Bandwidth, 1 for Energy</ResponseField>
        <ResponseField name="buy_type" type="string">Resource purchase method: 0 for System Estimate, 1 for Specified Amount</ResponseField>
        <ResponseField name="net_num" type="string">Bandwidth amount for delegation</ResponseField>
        <ResponseField name="energy_num" type="long">Energy amount for delegation</ResponseField>
        <ResponseField name="net_txid" type="long">Bandwidth delegation transaction txid</ResponseField>
        <ResponseField name="energy_txid" type="string">Energy delegation transaction txid</ResponseField>
        <ResponseField name="net_price" type="string">Bandwidth price</ResponseField>
        <ResponseField name="energy_price" type="string">Energy price</ResponseField>
        <ResponseField name="status" type="integer"> 0 for order creation success, 1 for resource delegation success, 2 for resource delegation failure</ResponseField>
    </Expandable>
</ResponseField>
