---
title: "Estimated Mining Fee For Transfer"
api: "GET /v2/chain/minerfee"
description: "Estimated mining fee for current coin transfer"
---

#### 请求参数数据结构
<ParamField query="time" type="long" required>
    Current timestamp, e.g.: `1600128590`
</ParamField>
<ParamField query="charset" type="string" required>
    Encoding format, typically `utf-8`
</ParamField>
<ParamField query="version" type="string" required>
    Version format, typically `v2`
</ParamField>
<ParamField query="symbol" type="string" required>
    Unique identifier for the coin, used for transfers, e.g.：`USDTERC20`
</ParamField>

#### 响应参数数据结构
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="feeSymbol" type="string">Fee coin, e.g. BSC</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, e.g.：`USETERC20`</ResponseField>
        <ResponseField name="fee" type="string">Fee, e.g. `0.00001`</ResponseField>
        <ResponseField name="type" type="integer">Type, `0` main chain coin transfer, `1` token transfer</ResponseField>
        <ResponseField name="ifCollection" type="bool">Whether the coin supports receive, `true` supports it, `false` does not support it</ResponseField>
    </Expandable>
</ResponseField>
