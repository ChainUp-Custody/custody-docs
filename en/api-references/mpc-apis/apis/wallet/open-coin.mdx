#### Response Data Parameters
<ResponseField name="open_main_chain" type="object[ ]">
    Opened networks in the workspace
    <Expandable title="object">
        <ResponseField name="coin_net" type="string">Coin network, e.g.：`MATIC`</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, e.g.：`MATIC1`</ResponseField>
        <ResponseField name="symbol_alias" type="string">Real name of the coin, e.g.：`MATIC`</ResponseField>
        <ResponseField name="support_acceleration" type="bool">Indicates if acceleration is supported (true/false)</ResponseField>
    </Expandable>
</ResponseField>
<ResponseField name="support_main_chain" type="object[ ]">
    Supported networks in MPC
    <Expandable title="object">
        <ResponseField name="coin_net" type="string">Coin network, e.g.：`ETH`</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, e.g.：`ETH`</ResponseField>
        <ResponseField name="symbol_alias" type="string">Real name of the coin, e.g.：`ETH`</ResponseField>
        <ResponseField name="support_acceleration" type="bool">Indicates if acceleration is supported (true/false)</ResponseField>
        <ResponseField name="if_open_chain" type="bool">Indicates if the network is opened (false/true)</ResponseField>
    </Expandable>
</ResponseField>
