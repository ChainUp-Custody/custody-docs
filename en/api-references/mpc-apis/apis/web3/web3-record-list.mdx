---
title: Get Web3 Transaction Records
api: GET /api/mpc/web3/trans_list
description: "Get all Web3 transaction records under a wallet, maximum of 100 records."
---

#### Request Parameters
<ParamField query="ids" type="string" required>
    Comma-separated `request_id` strings
</ParamField>



#### Response Data Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">Web3 transaction ID</ResponseField>
        <ResponseField name="request_id" type="string">Unique identifier for the transaction</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">Sub wallet id</ResponseField>
        <ResponseField name="txid" type="string">Transaction hash</ResponseField>
        <ResponseField name="symbol" type="string">Transaction coin</ResponseField>
        <ResponseField name="main_chain_symbol" type="string">Main chain coin symbol (unique identifier), e.g.：`ETH`</ResponseField>
        <ResponseField name="amount" type="string">Transaction amount. -1 indicates infinite</ResponseField>
        <ResponseField name="fee_symbol" type="string">Fee currency, e.g.：ETH</ResponseField>
        <ResponseField name="fee" type="string">Fee, e.g.：0.00123</ResponseField>
        <ResponseField name="real_fee" type="string">Actual fee consumed, e.g.：0.00111</ResponseField>
        <ResponseField name="created_at" type="long">Creation timestamp</ResponseField>
        <ResponseField name="updated_at" type="long">Modified timestamp</ResponseField>
        <ResponseField name="address_from" type="string">Transaction source address</ResponseField>
        <ResponseField name="address_to" type="string">Transaction destination address</ResponseField>
        <ResponseField name="interactive_contract" type="string">Interactive contract</ResponseField>
        <ResponseField name="confirmations" type="integer">Confirmations, e.g.：10</ResponseField>
        <ResponseField name="input_data" type="string">Hexadecimal data for contract transaction</ResponseField>
        <ResponseField name="status" type="integer">Transaction status: 1100 Pending Signature, 1200 Payment Processing, 2000 Payment Complete, 2100 Approval Rejected, 2200 Rejected, 2300 Transaction Discarded, 2400 Payment Failed</ResponseField>
        <ResponseField name="trans_source" type="integer">Transaction type： `1`app, `2`openapi</ResponseField>
        <ResponseField name="tx_height" type="long">Block height at which the transaction is completed</ResponseField>
    </Expandable>
</ResponseField>
