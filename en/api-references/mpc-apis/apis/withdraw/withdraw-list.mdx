---
title: Get Transfer Records
api: GET /api/mpc/billing/withdraw_list
description: "Get all wallet transfer records after a specific transaction ID for all wallets under the workspace"
---

#### Request Parameters
<ParamField query="ids" type="string" required>
    many `request_id` string, separated by commas, return up to 100 records.
</ParamField>



#### Response Data Parameters
<ResponseField type="object[ ]">
    <Expandable title="object">
        <ResponseField name="id" type="integer">transfer id</ResponseField>
        <ResponseField name="request_id" type="string">The unique identifier for transferring coins</ResponseField>
        <ResponseField name="sub_wallet_id" type="integer">wallet id</ResponseField>
        <ResponseField name="symbol" type="string">Unique identifier for the coin, used for transfers, e.g.：`USDTERC20`</ResponseField>
        <ResponseField name="amount" type="string">transfer amount</ResponseField>
        <ResponseField name="fee_symbol" type="string">gas fee coins, e.g.：ETH</ResponseField>
        <ResponseField name="fee" type="string">gas fee, e.g.：0.00123</ResponseField>
        <ResponseField name="real_fee" type="string">The actual gas fee consumed, e.g.：0.00111</ResponseField>
        <ResponseField name="created_at" type="long">Creation time timestamp</ResponseField>
        <ResponseField name="updated_at" type="long">Modification time timestamp</ResponseField>
        <ResponseField name="address_from" type="string">from account address</ResponseField>
        <ResponseField name="address_to" type="string">to account address</ResponseField>
        <ResponseField name="memo" type="string">to account address memo</ResponseField>
        <ResponseField name="txid" type="string">Transaction hash</ResponseField>
        <ResponseField name="confirmations" type="integer">Number of block confirmations, e.g.：10</ResponseField>
        <ResponseField name="status" type="integer">`1000` Awaiting approval, `1010` KYT Verification In Progress, `1100` Approved while awaiting signature, `1020` Pending KYT Manual Review, `1200` Payment in progress, `2000` Payment completed, `2100` Not Approved, `2101` KYT Verification Rejected, `2200` Rejected, `2300` Transaction discarded, `2400`Payment failed. [Details refer to](/help-center/manage-mpc/transaction/transfer#ii)</ResponseField>
        <ResponseField name="withdraw_source" type="integer">Transfer Type： `1`app, `2`openapi, `3`web</ResponseField>
        <ResponseField name="base_symbol" type="string">The unique identifier of the network of the coin to be transferred, e.g.：ETH</ResponseField>
        <ResponseField name="contract_address" type="string">transfer coin's contract address</ResponseField>
        <ResponseField name="tx_height" type="long">Block height at which the transaction is completed</ResponseField>
    </Expandable>
</ResponseField>
